pte_demand_manpower = {
	category = 6
	power_projection = no
	power_cost_base = 0.5
	prestige_base = 0.667
	ae_base = 0.168
	warscore_cost = {
		all_provinces = 0.0
		no_provinces = 15.0
		owner_keeps = 0.0
		conquered = 0.0
		returned_core = 0.0
		released_nation = 0.0
		cancelled_subject = 0.0
		concede_colonial_region = 0.0
	}
	warscore_cap = -1
	requires_demand_independence = no
	is_make_subject = no
	requires_is_allowed = no

	is_visible = {
		NOT = { has_country_modifier = monstrous_nation }
		OR = {
			FROM = {
				has_country_modifier = monstrous_nation
			}
			is_nomad = yes
			#tag = ROM
			has_reform = emperor_of_the_revolution_reform
			has_reform = military_dictatorship_reform
			has_reform = battle_pope_reform
			has_reform = pirate_republic_reform
			government = native
			government = tribal
			is_revolutionary = yes
			is_revolution_target = yes
		}
		FROM = {
			max_manpower = 1
		}
	}
	is_allowed = {
		OR = {
			AND = {
				FROM = {
					NOT = { has_country_modifier = monstrous_nation }
				}
				army_tradition = 5
			}
			AND = {
				prestige = 15
				OR = {
					FROM = {
						has_country_modifier = monstrous_nation
					}
					is_rival = FROM
				}
			}
		}
		FROM = {
			manpower_percentage = 0.1
		}
	}
	effect = {
		if {
			limit {
				FROM = {
					NOT = { has_country_modifier = monstrous_nation }
				}
			}
			add_army_tradition = -5
		}
		else {
			add_prestige = -20
		}
		export_to_variable = {
			which = enemy_max_manpower
			value = max_manpower
			who = FROM
		}
		divide_variable = { which = enemy_max_manpower value = 10 }
		divide_variable = { which = enemy_max_manpower value = 100 }
		multiply_variable = { which = enemy_max_manpower value = 100 }
		while = {
			limit = {
				check_variable = { which = enemy_max_manpower value = 1000 }
			}
			change_variable = { which = enemy_max_manpower value = -1000 }
			add_manpower = 1000
			FROM = {
				add_manpower = -1000
			}
		}
		while = {
			limit = {
				check_variable = { which = enemy_max_manpower value = 100 }
			}
			change_variable = { which = enemy_max_manpower value = -100 }
			add_manpower = 100
			FROM = {
				add_manpower = -100
			}
		}
		while = {
			limit = {
				check_variable = { which = enemy_max_manpower value = 10 }
			}
			change_variable = { which = enemy_max_manpower value = -10 }
			add_manpower = 10
			FROM = {
				add_manpower = -10
			}
		}
		while = {
			limit = {
				check_variable = { which = enemy_max_manpower value = 1 }
			}
			change_variable = { which = enemy_max_manpower value = -1 }
			add_manpower = 1
			FROM = {
				add_manpower = -1
			}
		}
		while = {
			limit = {
				check_variable = { which = enemy_max_manpower value = 0.1 }
			}
			change_variable = { which = enemy_max_manpower value = -0.1 }
			add_manpower = 0.1
			FROM = {
				add_manpower = -0.1
			}
		}
		set_variable = { which = enemy_max_manpower value = 0 }
	}
	ai_weight = {
		export_to_variable = {
			variable_name = ai_value
			value = 1
		}
		limit = {
			NOT = { manpower_percentage = 0.1 }
			FROM = {
				max_manpower = 50
				NOT = {
					num_of_owned_provinces_with = {
						value = 3
						OR = {
							is_claim = ROOT
							is_core = ROOT
						}
					}
				}
			}
			OR = {
				border_distance = {
					who = FROM
					distance = 300
				}
				FROM = {
					num_of_cities = 5
				}
			}
		}
	}
}
